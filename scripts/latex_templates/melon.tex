% 青提甜瓜风格
% 清新自然的绿色调，适合生活方式类内容

\documentclass[preview,border=0pt]{standalone}
\usepackage{ctex}
\usepackage{tikz}
\usepackage{graphicx}

% 配色定义
\definecolor{background}{HTML}{E8F5E9}  % 浅绿
\definecolor{primary}{HTML}{2E7D32}     % 深绿
\definecolor{secondary}{HTML}{81C784}   % 中绿
\definecolor{accent}{HTML}{C8E6C9}      % 淡绿

% 尺寸 (1080x1440 px at 100 dpi = 10.8x14.4 cm)
\newlength{\coverwidth}
\newlength{\coverheight}
\setlength{\coverwidth}{10.8cm}
\setlength{\coverheight}{14.4cm}

\begin{document}
\begin{tikzpicture}
  % 背景
  \fill[background] (0,0) rectangle (\coverwidth,\coverheight);

  % 左侧装饰条
  \fill[secondary, rounded corners=3pt] (0.3cm,1cm) rectangle (0.6cm,13.4cm);

  % 顶部圆点装饰
  \foreach \y in {12.5, 12.2, 11.9} {
    \fill[secondary] (9.5cm,\y cm) circle (0.15cm);
  }

  % 主标题
  \node[font=\fontsize{48pt}{56pt}\selectfont\bfseries,
        text=primary,
        align=center,
        text width=8cm]
    at (5.6cm,10cm) {主标题};

  % 副标题 (带高亮背景)
  \node[font=\fontsize{32pt}{40pt}\selectfont,
        text=primary,
        fill=accent,
        rounded corners=8pt,
        inner sep=12pt]
    at (5.4cm,8cm) {副标题};

  % 装饰文字
  \node[font=\fontsize{18pt}{24pt}\selectfont,
        text=secondary]
    at (5.4cm,5cm) {附加信息};

  % 底部装饰线
  \draw[secondary, line width=2pt] (1cm,1.5cm) -- (9.8cm,1.5cm);
\end{tikzpicture}
\end{document}
