\documentclass[preview,border=0pt]{standalone}
\usepackage{ctex}
\usepackage{tikz}
\usepackage{graphicx}

% 配色定义
\definecolor{background}{HTML}{E3F2FD}  % 浅蓝
\definecolor{primary}{HTML}{1565C0}     % 深蓝
\definecolor{secondary}{HTML}{42A5F5}   % 中蓝
\definecolor{accent}{HTML}{90CAF9}      % 淡蓝

% 尺寸 (1080x1440 px at 100 dpi = 10.8x14.4 cm)
\newlength{\coverwidth}
\newlength{\coverheight}
\setlength{\coverwidth}{10.8cm}
\setlength{\coverheight}{14.4cm}

\begin{document}
\begin{tikzpicture}
  % 背景
  \fill[background] (0,0) rectangle (\coverwidth,\coverheight);

  % 顶部波浪装饰
  \fill[accent] (0,12.5cm) --
    (2.7cm,13cm) -- (5.4cm,12.7cm) --
    (8.1cm,13.2cm) (10.8cm,12.8cm) --
    (10.8cm,14.4cm) -- (0,14.4cm) -- cycle;

  % 主标题
  \node[font=\fontsize{48pt}{56pt}\selectfont\bfseries,
        text=primary,
        align=center,
        text width=9cm]
    at (5.4cm,10cm) {主标题};

  % 副标题 (带高亮背景)
  \node[font=\fontsize{32pt}{40pt}\selectfont,
        text=primary,
        fill=secondary,
        rounded corners=8pt,
        inner sep=12pt]
    at (5.4cm,8cm) {副标题};

  % 装饰文字
  \node[font=\fontsize{18pt}{24pt}\selectfont,
        text=primary!70]
    at (5.4cm,5cm) {附加信息};

  % 底部波浪装饰
  \fill[accent] (0,0) --
    (2.7cm,0.8cm) -- (5.4cm,0.5cm) --
    (8.1cm,1cm) -- (10.8cm,0.6cm) --
    (10.8cm,0) -- cycle;
\end{tikzpicture}
\end{document}
